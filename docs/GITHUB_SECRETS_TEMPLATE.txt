# GitHub Secrets Template for Solana Trading Bot
# ==============================================
# 
# Copy each value into GitHub → Settings → Secrets and variables → Actions
# Create a new "Repository secret" for each item below.
#
# HOW TO ADD SECRETS:
# 1. Go to your GitHub repo
# 2. Click Settings → Secrets and variables → Actions
# 3. Click "New repository secret"
# 4. Add each secret below

# ============================================
# EC2 CONNECTION SECRETS
# ============================================

# EC2_HOST
# Your EC2 public IP address or DNS
# Example: 54.123.45.67 or ec2-54-123-45-67.compute-1.amazonaws.com
EC2_HOST=YOUR_EC2_PUBLIC_IP

# EC2_USER
# SSH username (usually 'ubuntu' for Ubuntu AMIs)
EC2_USER=ubuntu

# EC2_SSH_PRIVATE_KEY
# Your SSH private key content (the .pem file contents)
# Copy the ENTIRE contents of your .pem file including:
# -----BEGIN RSA PRIVATE KEY-----
# ... key content ...
# -----END RSA PRIVATE KEY-----
#
# IMPORTANT: When pasting in GitHub, paste the complete key including headers
EC2_SSH_PRIVATE_KEY=PASTE_YOUR_PEM_FILE_CONTENTS_HERE

# ============================================
# SOLANA CONFIGURATION
# ============================================

# SOLANA_RPC_URL
# Your Solana RPC endpoint
# Devnet (testing): https://api.devnet.solana.com
# Mainnet: Get from Helius (https://helius.dev) or QuickNode
SOLANA_RPC_URL=https://api.devnet.solana.com

# DRIFT_ENV
# 'devnet' for testing, 'mainnet-beta' for production
DRIFT_ENV=devnet

# WALLET_PRIVATE_KEY
# Base58 encoded private key (NEVER share this!)
# Generate with: solana-keygen new
WALLET_PRIVATE_KEY=YOUR_BASE58_PRIVATE_KEY

# ============================================
# AZURE OPENAI (GitHub Enterprise)
# ============================================

# AZURE_OPENAI_API_KEY
# Your Azure OpenAI API key (from GitHub Enterprise / Azure portal)
AZURE_OPENAI_API_KEY=YOUR_AZURE_OPENAI_KEY

# AZURE_OPENAI_ENDPOINT
# Your Azure OpenAI endpoint URL
# Example: https://your-resource-name.openai.azure.com/
AZURE_OPENAI_ENDPOINT=https://YOUR_RESOURCE.openai.azure.com/

# AZURE_OPENAI_DEPLOYMENT
# Your deployment name (model deployment in Azure)
AZURE_OPENAI_DEPLOYMENT=gpt-4o

# AZURE_OPENAI_API_VERSION
# API version to use
AZURE_OPENAI_API_VERSION=2024-02-15-preview

# LLM_MODEL
# Model name (usually matches deployment name)
LLM_MODEL=gpt-4o

# ============================================
# APPLICATION SETTINGS
# ============================================

# CHECK_INTERVAL_SECONDS
# How often to check price conditions (in seconds)
CHECK_INTERVAL_SECONDS=10

# LOG_LEVEL
# Logging level: DEBUG, INFO, WARNING, ERROR
LOG_LEVEL=INFO

# APP_SECRET_KEY
# Random string for session security (generate with: openssl rand -hex 32)
APP_SECRET_KEY=YOUR_RANDOM_32_CHAR_STRING
